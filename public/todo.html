grb.ws_blob('/todo', function (blob, object) {
    if (!blob.read('list')) {
        blob.create('list', {});
    }

    input.onkeydown = function(e) {
        if (e.keyCode == 13) {
            blob.create("list." + input.value, input.value);
            input.value = "";
        }
    };

    blob.on('create', function(p, k, v) {
        var item = document.createElement('li');
        item.textContent = v;
        items.insertBefore(item, input);
    });
});
